projects/congarevenuecloud/elements/src/lib/product-compare/product-compare.component.ts

Description

The Product Compare component is used to display a side by side comparison of product records. Each product is displayed as a card at the top, with a table below each card that lists its features.

import { ProductCompareModule } from '@congarevenuecloud/elements';

imports: [ProductCompareModule, ...]
})
export class AppModule {}
